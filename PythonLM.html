    """Returns a string describing the
    length of the sequences into 
    empty, very long, normal and short"""
    if len(s) == 0:
        ans = 'empty'
    elif len(s) > 10:
        ans = 'very long'
    elif len(s) > 7:
        ans = 'normal'
    else:
        ans = 'short'
    return ans
